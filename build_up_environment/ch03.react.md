main.tsx 파일

```typescriptreact
import ReactDOM from "react-dom/client";
import App from "./App";

function main() {
    const element = document.getElementById('root');

    if(!element) {
        return;
    }
    const root = ReactDOM.createRoot(element);
    root.render(<App/>);
}

main();
```
